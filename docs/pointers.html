int addition (int a, int b)
{ return (a+b); }

int subtraction (int a, int b)
{ return (a-b); }

int (*minus)(int,int) = subtraction;

int operation (int x, int y, int (*functocall)(int,int))
{ int g;
g = (*functocall)(x,y);
return (g); }

int main () {
int m,n;
m = operation (7, 5, addition);
n = operation (20, m, minus);

Σύνοψη τελεστών για λειτουργίες δεικτών

*xδείκτης στο x (δήλωση), περιεχόμενο του x (έκφραση)
&xδιεύθυνση του x (έκφραση), αναφορά (σε πέρασμα)
x.yμέλος y του αντικειμένου x
(*x).yμέλος y του αντικειμένου που δεικτοδοτείται από τον x
x->yισοδύναμο με το προηγούμενο
x[0]το πρώτο αντικείμενο που δεικτοδοτείται από τον
x[n]το n+1 αντικείμενο που δεικτοδοτείται από τον x
